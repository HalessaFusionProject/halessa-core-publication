import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# Define spatial grid
x = np.linspace(-2, 2, 100)
y = np.linspace(-2, 2, 100)
x, y = np.meshgrid(x, y)

# Define charge density distribution rho(x, y)
def charge_density(x, y):
    return np.exp(-x**2 - y**2)  # Gaussian distribution centered at (0,0)

# Define rotational field potential Phi(x, y)
def rotational_potential(x, y):
    return np.sin(np.pi * x) * np.cos(np.pi * y)  # Sample topological rotation

# Define harmonic feedback frequency psi(x, y)
def harmonic_feedback(x, y):
    return np.cos(2 * np.pi * x) * np.sin(2 * np.pi * y)  # Feedback loop behavior

# Symmetry collapse coefficient (assumed constant for simplicity)
sigma = 0.8

# Compute Q(x, y)
rho = charge_density(x, y)
phi = rotational_potential(x, y)
psi = harmonic_feedback(x, y)

Q = sigma * rho * phi * psi  # Pointwise computation

# Plot the energy distribution Q
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(x, y, Q, cmap='plasma')
ax.set_title('Harmonic Fusion Energy Output Q(x, y)')
ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_zlabel('Energy Density Q')

plt.tight_layout()
plt.show()
