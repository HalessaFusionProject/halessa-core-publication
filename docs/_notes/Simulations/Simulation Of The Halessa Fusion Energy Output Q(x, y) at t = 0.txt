# Re-run after kernel reset: simulate and visualize Halessa Fusion Energy Output

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

# Define spatial grid
x = np.linspace(-2, 2, 100)
y = np.linspace(-2, 2, 100)
x, y = np.meshgrid(x, y)
t = 0  # fixed time for static simulation

# Charge density function
def charge_density(x, y):
    rho_0 = 1.0
    sigma_rho = 0.8
    return rho_0 * np.exp(-(x**2 + y**2) / (2 * sigma_rho**2))

# Rotational potential function
def rotational_potential(x, y):
    return np.sin(np.pi * x) * np.cos(np.pi * y)

# Harmonic feedback function
def harmonic_feedback(x, y, t):
    omega = 1.0
    return np.cos(2 * np.pi * x + omega * t) * np.sin(2 * np.pi * y + omega * t)

# Symmetry collapse coefficient
sigma = 0.8

# Compute Q(x, y)
rho = charge_density(x, y)
phi = rotational_potential(x, y)
psi = harmonic_feedback(x, y, t)
Q = sigma * rho * phi * psi

# Plot Q(x, y)
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')
ax.plot_surface(x, y, Q, cmap='inferno')
ax.set_title('Halessa Fusion Energy Output Q(x, y) at t=0')
ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_zlabel('Energy Density Q')

plt.tight_layout()
plt.show()
