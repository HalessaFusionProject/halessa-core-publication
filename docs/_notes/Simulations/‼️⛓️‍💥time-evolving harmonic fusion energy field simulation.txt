# Re-initialize everything after reset and rerun the animation for dynamic harmonic fusion energy

import numpy as np
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
from mpl_toolkits.mplot3d import Axes3D

# Spatial grid
x = np.linspace(-2, 2, 100)
y = np.linspace(-2, 2, 100)
x, y = np.meshgrid(x, y)

# Charge density function
def charge_density(x, y):
    rho_0 = 1.0
    sigma_rho = 0.8
    return rho_0 * np.exp(-(x**2 + y**2) / (2 * sigma_rho**2))

# Rotational potential
def rotational_potential(x, y):
    return np.sin(np.pi * x) * np.cos(np.pi * y)

# Harmonic feedback with time
def harmonic_feedback(x, y, t):
    omega = 1.0
    return np.cos(2 * np.pi * x + omega * t) * np.sin(2 * np.pi * y + omega * t)

# Constants
sigma = 0.8
time_frames = 60
time_array = np.linspace(0, 2 * np.pi, time_frames)

# Setup plot
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')
ax.set_zlim(-0.8, 0.8)

# Update function for animation
def update(frame):
    ax.clear()
    rho = charge_density(x, y)
    phi = rotational_potential(x, y)
    psi = harmonic_feedback(x, y, time_array[frame])
    Q = sigma * rho * phi * psi
    surf = ax.plot_surface(x, y, Q, cmap='plasma')
    ax.set_title(f'Harmonic Fusion Energy Q(x, y, t={time_array[frame]:.2f})')
    ax.set_xlabel('X')
    ax.set_ylabel('Y')
    ax.set_zlabel('Energy Density Q')
    ax.set_zlim(-0.8, 0.8)
    return surf,

# Create animation
anim = FuncAnimation(fig, update, frames=time_frames, interval=100)
plt.close(fig)
